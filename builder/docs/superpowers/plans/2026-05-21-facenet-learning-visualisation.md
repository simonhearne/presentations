# FaceNet Learning Visualisation Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build an ambient three.js visualisation for slide 5 of `talks/vector-search-visualised/` showing FaceNet turning face images into vectors and learning the embedding space via triplet loss.

**Architecture:** One deck-folder three.js module, `facenet-learning.js`, built up in layers (embedding space → network + travellers → triplet events). It embeds celebrity faces as base64 data URIs produced by a generator script, `data/make_facenet_module.js`. The module is ambient (returns only `dispose`), so arrow keys pass through to deck navigation. The slide uses the existing `.three-bg` layout class — no CSS changes.

**Tech Stack:** three.js (via the deck's existing CDN importmap), the existing `three` frontmatter block and build-time module inlining, Node's built-in modules for the generator.

---

## Design reference

Full design: `docs/superpowers/specs/2026-05-21-facenet-learning-visualisation-design.md`.

Key conventions:
- The module is **inlined as base64 by the build**, so faces must be embedded as data URIs inside the file — relative `./data/faces/...` paths would not resolve at runtime (same as `cloud.js`).
- `facenet-learning.js` contains a `FACES` constant fenced by `/* FACES_START ... */` and `/* FACES_END */` sentinels. The generator rewrites **only** that region. Everything else is hand-written.
- Task 1 creates the skeleton and runs the generator once, so the committed module has real face data. Tasks 2–4 use **`Edit`** on stable anchors and never disturb the `FACES` block.
- No automated tests (browser/WebGL code + fs-bound generator — `cloud.js`/`connectome.js`/`make_cloud_module.js` precedent). Each task ends with a build and a visual review.

// talks/vector-search-visualised/facenet-learning.js
// Ambient visualisation of FaceNet: faces enter through a stylised network,
// morph into identity-coloured points, and triplet-loss events pull/push the
// embedding space. Ambient — returns only dispose(), no advance/retreat.

import * as THREE from 'three';

/* FACES_START — generated by data/make_facenet_module.js, do not edit by hand */
const FACES = [];
/* FACES_END */

const DEFAULTS = { background: '#070d18', speed: 1 };

// One cluster centre per identity. Length MUST equal FACES.length.
const CLUSTERS = [
  [-0.3, 1.2, 0.2],
  [2.3, 1.3, -0.5],
  [3.0, -0.9, 0.6],
  [0.4, -1.4, -0.2],
  [1.5, 0.0, 0.9],
];
const POINTS_PER_IDENTITY = 5;
const STRIP_X = -3.3; // network strip x position
const SPAWN_X = -7.2; // travellers enter just off the left edge

export default function init({ canvas, opts = {} }) {
  const cfg = { ...DEFAULTS, ...opts };
  const reduced = !!(
    window.matchMedia &&
    window.matchMedia('(prefers-reduced-motion: reduce)').matches
  );

  const renderer = new THREE.WebGLRenderer({ canvas, antialias: true, alpha: true });
  renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
  let w = canvas.clientWidth || canvas.width;
  let h = canvas.clientHeight || canvas.height;
  renderer.setSize(w, h, false);
  renderer.setClearColor(new THREE.Color(cfg.background), 1);

  const scene = new THREE.Scene();
  scene.fog = new THREE.Fog(new THREE.Color(cfg.background), 8, 18);

  const camera = new THREE.PerspectiveCamera(42, (w || 1) / (h || 1), 0.1, 100);
  camera.position.set(0.6, 0, 9.4);
  camera.lookAt(0.6, 0, 0);

  const group = new THREE.Group();
  scene.add(group);

  const disposables = []; // anything with a .dispose()
  const updaters = []; // fns (dt) => void, run each frame

  // Soft circular sprite so points render as round dots.
  function discTexture() {
    const s = 64;
    const cv = document.createElement('canvas');
    cv.width = cv.height = s;
    const g = cv.getContext('2d');
    const grd = g.createRadialGradient(s / 2, s / 2, 0, s / 2, s / 2, s / 2);
    grd.addColorStop(0, 'rgba(255,255,255,1)');
    grd.addColorStop(0.5, 'rgba(255,255,255,1)');
    grd.addColorStop(1, 'rgba(255,255,255,0)');
    g.fillStyle = grd;
    g.fillRect(0, 0, s, s);
    const tex = new THREE.CanvasTexture(cv);
    disposables.push(tex);
    return tex;
  }
  const sprite = discTexture();

  let pointsLayer = null;

  // === build layers ===

  let raf = 0;
  let last = performance.now();
  let elapsed = 0;
  function frame(now) {
    const dt = Math.min((now - last) / 1000, 0.05);
    last = now;
    const sdt = dt * cfg.speed;
    elapsed += sdt;
    group.rotation.y = reduced ? 0 : Math.sin(elapsed * 0.16) * 0.12;
    for (let i = 0; i < updaters.length; i++) updaters[i](sdt);
    renderer.render(scene, camera);
    raf = requestAnimationFrame(frame);
  }
  raf = requestAnimationFrame(frame);

  function onResize() {
    w = canvas.clientWidth || canvas.width;
    h = canvas.clientHeight || canvas.height;
    if (!w || !h) return;
    renderer.setSize(w, h, false);
    camera.aspect = w / h;
    camera.updateProjectionMatrix();
  }
  window.addEventListener('resize', onResize);

  return {
    dispose() {
      cancelAnimationFrame(raf);
      window.removeEventListener('resize', onResize);
      for (const d of disposables) d.dispose();
      renderer.dispose();
    },
  };
}
```

#!/usr/bin/env node
// Regenerates the FACES data block in ../facenet-learning.js.
//
// The face JPEGs are inlined as base64 data URIs because the build
// base64-encodes the whole module, so a relative ./data/faces/... path
// would not resolve at runtime. Only the region between the FACES_START
// and FACES_END sentinels is rewritten; the rest of the module is
// hand-written and left untouched.

import { readFileSync, writeFileSync } from 'node:fs';
import { dirname, resolve, join } from 'node:path';
import { fileURLToPath } from 'node:url';

const here = dirname(fileURLToPath(import.meta.url));
const facesDir = join(here, 'faces');
const modulePath = resolve(here, '..', 'facenet-learning.js');

// Curated subset: one face per identity, paired with an identity colour.
// Length and order MUST match CLUSTERS in facenet-learning.js.
const IDENTITIES = [
  { file: '003983.jpg', color: '#5b86e8' },
  { file: '016605.jpg', color: '#e8a14b' },
  { file: '020172.jpg', color: '#6dd6a0' },
  { file: '042926.jpg', color: '#e87da8' },
  { file: '063886.jpg', color: '#a98de8' },
];

const START =
  '/* FACES_START — generated by data/make_facenet_module.js, do not edit by hand */';
const END = '/* FACES_END */';

const entries = IDENTITIES.map(({ file, color }) => {
  const b64 = readFileSync(join(facesDir, file)).toString('base64');
  return `  { color: ${JSON.stringify(color)}, dataUri: "data:image/jpeg;base64,${b64}" }`;
});
const block = `${START}\nconst FACES = [\n${entries.join(',\n')}\n];\n${END}`;

const src = readFileSync(modulePath, 'utf8');
const startIdx = src.indexOf(START);
const endIdx = src.indexOf(END);
if (startIdx === -1 || endIdx === -1) {
  throw new Error('FACES sentinels not found in facenet-learning.js');
}
const next = src.slice(0, startIdx) + block + src.slice(endIdx + END.length);
writeFileSync(modulePath, next);
console.log(`wrote FACES block to ${modulePath} (${IDENTITIES.length} faces)`);
```

  // ---- layer: embedding space ----
  function buildPoints() {
    const count = FACES.length * POINTS_PER_IDENTITY;
    const positions = new Float32Array(count * 3);
    const colors = new Float32Array(count * 3);
    const baseColors = FACES.map((f) => new THREE.Color(f.color));
    const centres = CLUSTERS.map((c) => new THREE.Vector3(c[0], c[1], c[2]));
    const pool = [];

    for (let i = 0; i < count; i++) {
      const id = Math.floor(i / POINTS_PER_IDENTITY);
      const home = new THREE.Vector3(
        (Math.random() * 2 - 1) * 0.55,
        (Math.random() * 2 - 1) * 0.55,
        (Math.random() * 2 - 1) * 0.5,
      );
      const pos = centres[id].clone().add(home);
      pool.push({
        id,
        home,
        pos,
        phase: Math.random() * Math.PI * 2,
        triplet: new THREE.Vector3(), // transient offset, owned by the triplet layer
        flash: 0, // born-flash timer (seconds remaining)
      });
      pos.toArray(positions, i * 3);
      baseColors[id].toArray(colors, i * 3);
    }

    const geo = new THREE.BufferGeometry();
    geo.setAttribute('position', new THREE.BufferAttribute(positions, 3));
    geo.setAttribute('color', new THREE.BufferAttribute(colors, 3));
    const mat = new THREE.PointsMaterial({
      map: sprite,
      size: 0.36,
      sizeAttenuation: true,
      vertexColors: true,
      transparent: true,
      opacity: 0.96,
      depthWrite: false,
    });
    group.add(new THREE.Points(geo, mat));
    disposables.push(geo, mat);

    const wander = new THREE.Vector3();
    const target = new THREE.Vector3();
    const white = new THREE.Color(0xffffff);
    const tmpCol = new THREE.Color();

    function update(dt) {
      for (let i = 0; i < pool.length; i++) {
        const p = pool[i];
        if (reduced) {
          wander.set(0, 0, 0);
        } else {
          wander.set(
            Math.sin(elapsed * 0.5 + p.phase) * 0.16,
            Math.cos(elapsed * 0.43 + p.phase) * 0.16,
            Math.sin(elapsed * 0.37 + p.phase) * 0.13,
          );
        }
        target.copy(centres[p.id]).add(p.home).add(wander).add(p.triplet);
        p.pos.lerp(target, Math.min(1, dt * 2.2));
        p.pos.toArray(positions, i * 3);
        if (p.flash > 0) {
          p.flash = Math.max(0, p.flash - dt);
          tmpCol.copy(baseColors[p.id]).lerp(white, (p.flash / 0.6) * 0.85);
        } else {
          tmpCol.copy(baseColors[p.id]);
        }
        tmpCol.toArray(colors, i * 3);
      }
      geo.attributes.position.needsUpdate = true;
      geo.attributes.color.needsUpdate = true;
    }
    updaters.push(update);

    pointsLayer = {
      pool,
      centres,
      // Teleport a random point of identity `id` to `vec` and flash it white.
      spawnAt(id, vec) {
        const p = pool[id * POINTS_PER_IDENTITY + ((Math.random() * POINTS_PER_IDENTITY) | 0)];
        p.pos.copy(vec);
        p.flash = 0.6;
        return p;
      },
    };
  }

  return {

  // Small text label as a camera-facing sprite.
  function textSprite(text, color) {
    const pad = 10;
    const cv = document.createElement('canvas');
    let ctx = cv.getContext('2d');
    const font = '600 30px Inter, system-ui, sans-serif';
    ctx.font = font;
    const tw = Math.ceil(ctx.measureText(text).width);
    cv.width = tw + pad * 2;
    cv.height = 46;
    ctx = cv.getContext('2d');
    ctx.font = font;
    ctx.fillStyle = color;
    ctx.textBaseline = 'middle';
    ctx.fillText(text, pad, cv.height / 2);
    const tex = new THREE.CanvasTexture(cv);
    tex.colorSpace = THREE.SRGBColorSpace;
    const mat = new THREE.SpriteMaterial({
      map: tex,
      transparent: true,
      depthWrite: false,
      depthTest: false,
    });
    const sp = new THREE.Sprite(mat);
    sp.scale.set((cv.width / cv.height) * 0.6, 0.6, 1);
    disposables.push(tex, mat);
    return sp;
  }

  // ---- layer: network strip ----
  function buildNetwork() {
    const cols = 3;
    const rows = 4;
    const nodePos = new Float32Array(cols * rows * 3);
    let k = 0;
    for (let c = 0; c < cols; c++) {
      for (let r = 0; r < rows; r++) {
        nodePos[k++] = STRIP_X + c * 0.34;
        nodePos[k++] = (r - (rows - 1) / 2) * 0.46;
        nodePos[k++] = 0;
      }
    }
    const ngeo = new THREE.BufferGeometry();
    ngeo.setAttribute('position', new THREE.BufferAttribute(nodePos, 3));
    const nmat = new THREE.PointsMaterial({
      map: sprite,
      color: 0x5b86e8,
      size: 0.2,
      sizeAttenuation: true,
      transparent: true,
      opacity: 0.5,
      depthWrite: false,
    });
    group.add(new THREE.Points(ngeo, nmat));
    disposables.push(ngeo, nmat);

    const segs = [];
    for (let c = 0; c < cols - 1; c++) {
      for (let r1 = 0; r1 < rows; r1++) {
        for (let r2 = 0; r2 < rows; r2++) {
          segs.push(
            STRIP_X + c * 0.34,
            (r1 - (rows - 1) / 2) * 0.46,
            0,
            STRIP_X + (c + 1) * 0.34,
            (r2 - (rows - 1) / 2) * 0.46,
            0,
          );
        }
      }
    }
    const egeo = new THREE.BufferGeometry();
    egeo.setAttribute('position', new THREE.BufferAttribute(new Float32Array(segs), 3));
    const emat = new THREE.LineBasicMaterial({
      color: 0x5b86e8,
      transparent: true,
      opacity: 0.12,
      depthWrite: false,
    });
    group.add(new THREE.LineSegments(egeo, emat));
    disposables.push(egeo, emat);

    const label = textSprite('network → vector', '#7f93c8');
    label.position.set(STRIP_X + 0.34, -1.5, 0);
    label.scale.multiplyScalar(0.82);
    group.add(label);
  }

  // ---- layer: travellers (image → vector) ----
  function buildTravellers() {
    const loader = new THREE.TextureLoader();
    const faceTex = FACES.map((f) => {
      const t = loader.load(f.dataUri);
      t.colorSpace = THREE.SRGBColorSpace;
      disposables.push(t);
      return t;
    });
    const geo = new THREE.PlaneGeometry(1, 1.2);
    disposables.push(geo);

    const entry = new THREE.Vector3(STRIP_X, 0, 0);
    const travellers = [];
    for (let i = 0; i < 2; i++) {
      const mat = new THREE.MeshBasicMaterial({ transparent: true, depthWrite: false });
      const mesh = new THREE.Mesh(geo, mat);
      mesh.visible = false;
      group.add(mesh);
      disposables.push(mat);
      travellers.push({ mesh, mat, state: 'idle', t: 0, id: 0, morphed: false });
    }
    let spawnTimer = 1.0;

    function launch(tr) {
      tr.id = (Math.random() * FACES.length) | 0;
      tr.mat.map = faceTex[tr.id];
      tr.mat.needsUpdate = true;
      tr.mesh.position.set(SPAWN_X, (Math.random() * 2 - 1) * 1.4, 0);
      tr.mesh.scale.set(1, 1, 1);
      tr.mat.opacity = 1;
      tr.mesh.visible = true;
      tr.state = 'enter';
      tr.t = 0;
      tr.morphed = false;
    }

    function update(dt) {
      if (reduced) return;
      spawnTimer -= dt;
      if (spawnTimer <= 0) {
        const idle = travellers.find((t) => t.state === 'idle');
        if (idle) launch(idle);
        spawnTimer = 3.4 + Math.random() * 2.2;
      }
      for (const tr of travellers) {
        if (tr.state === 'idle') continue;
        tr.t += dt;
        if (tr.state === 'enter') {
          tr.mesh.position.lerp(entry, Math.min(1, dt * 1.6));
          if (tr.mesh.position.distanceTo(entry) < 0.15) {
            tr.state = 'process';
            tr.t = 0;
          }
        } else if (tr.state === 'process') {
          if (tr.t > 0.6) {
            tr.state = 'morph';
            tr.t = 0;
          }
        } else if (tr.state === 'morph') {
          if (!tr.morphed) {
            pointsLayer.spawnAt(tr.id, tr.mesh.position.clone());
            tr.morphed = true;
          }
          const m = Math.min(1, tr.t / 0.5);
          tr.mesh.scale.setScalar(1 - m * 0.8);
          tr.mat.opacity = 1 - m;
          if (m >= 1) {
            tr.mesh.visible = false;
            tr.state = 'idle';
          }
        }
      }
    }
    updaters.push(update);
  }

  return {

    const ringGeo = new THREE.RingGeometry(0.34, 0.4, 40);

      color: 0xffffff,

    function link(color) {
      const g = new THREE.BufferGeometry();
      g.setAttribute('position', new THREE.BufferAttribute(new Float32Array(6), 3));
      const m = new THREE.LineBasicMaterial({
        color,
        transparent: true,
        opacity: 0,
        depthWrite: false,
      });
      group.add(new THREE.Line(g, m));
      disposables.push(g, m);
      return { g, m };
    }
    const pull = link(0x6dd6a0);
    const push = link(0xe86d6d);

    const pullLabel = textSprite('pull · same identity', '#6dd6a0');
    const pushLabel = textSprite('push · different', '#e86d6d');
    pullLabel.material.opacity = 0;
    pushLabel.material.opacity = 0;
    group.add(pullLabel);
    group.add(pushLabel);

    const LABEL_UP = new THREE.Vector3(0, 0.34, 0);
    const DURATION = 3.0;
    const dir = new THREE.Vector3();
    const mid = new THREE.Vector3();
    let clock = reduced ? Infinity : 1.5; // seconds until the next event
    let ev = null;

    function setLine(L, a, b) {
      const p = L.g.attributes.position.array;
      a.toArray(p, 0);
      b.toArray(p, 3);
      L.g.attributes.position.needsUpdate = true;
    }

    function update(dt) {
      const pool = pointsLayer.pool;
      if (!ev) {
        clock -= dt;
        if (clock <= 0) {
          const anchorId = (Math.random() * FACES.length) | 0;
          let negId = (Math.random() * FACES.length) | 0;
          if (negId === anchorId) negId = (negId + 1) % FACES.length;
          const base = anchorId * POINTS_PER_IDENTITY;
          const ai = (Math.random() * POINTS_PER_IDENTITY) | 0;
          let pi = (Math.random() * POINTS_PER_IDENTITY) | 0;
          if (pi === ai) pi = (pi + 1) % POINTS_PER_IDENTITY;

            t: 0,
            anchor: pool[base + ai],
            positive: pool[base + pi],
            negative: pool[negId * POINTS_PER_IDENTITY + ((Math.random() * POINTS_PER_IDENTITY) | 0)],
          };

        for (let i = 0; i < pool.length; i++) {
          pool[i].triplet.multiplyScalar(Math.max(0, 1 - dt * 2.5));

        return;
